def download_youtube_video(video_url):
    ydl_opts = {
        # Keeps your 480p target
        'format': 'bestvideo[height<=480]+bestaudio/best[height<=480]',
        'restrictfilenames': True,
        'outtmpl': '%(title)s_480p.%(ext)s',
        
        # --- THE FIXES FOR THE BOT BLOCK ---
        # 1. Force yt-dlp to use alternative streaming clients (like TV)
        'extractor_args': {
            'youtube': {
                'player_client': ['tv', 'web'],
            }
        },
        # 2. Prevent YouTube from serving broken cached sessions
        'rm_cache_dir': True,
        # 3. Use an alternative fallback downloader strategy if it hits a 403 error
        'check_formats': True,
    }
    
    print("\nStarting download... Overriding player clients to bypass block.")
    
    try:
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            ydl.download([video_url])
        print("\nDownload completed successfully!")
    except Exception as e:
        print(f"\nAn error occurred: {e}")

if __name__ == "__main__":
    url = input("Enter video url: ")
